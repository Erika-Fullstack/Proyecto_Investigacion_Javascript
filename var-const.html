Variables y Constantes

Ejemplo Indefinido (Undefined)
                            
                    // Declaramos una variable Undefined
                    let userName;
                                
                    // Llamamos a la variable
                    console.log(userName); // salida: undefined

Cuando una variable se ha declarado pero no se le ha asignado un valor.

Ejemplo Nulo (Null)
                      
                    // Declaramos una variable Null
                    let selectedProduct = null;
                        
                    // Llamamos a la variable
                    console.log(selectedProduct); // salida: null

Indica que una variable no tiene valor o está vacía .

Ejemplo Cadena (String)
                      
                    // Declaramos el valor de la variable String
                    userName = "Carla Siles";
                        
                    // Llamamos a la variable
                    console.log(userName); // salida: "Carla Siles"

                    //otra forma de llamarlo
                    typeof(userName); salida: "carla siles"

Contiene secuencias de caracteres, como texto.

Ejemplo Funciones (Function)
                      
                    // Declaramos una variable Function
                    let greet = function (name) {
                    return `Hola, ${name}!`; };

                    // Llamamos a la variable
                    console.log(greet("Carla")); // salida: "Hola, Carla!"

En este ejemplo, estamos asignando a una variable un bloque de código que realiza una operación específica. Esto significa que la variable no solo puede almacenar datos primitivos como números o cadenas, sino también funcionalidad, permitiéndonos invocar esa funcionalidad cuando sea necesario usando la variable..

Ejemplo Número
                      
                    // Declaramos variables Number
                    let age = 29;
                    let height = 1.60;
                        
                    // Llamamos a las variables
                    console.log(age); // salida: 29
                    console.log(height); // salida: 1.6

                    //otra forma de llamarlo:
                    typeof(age); "number"

Variable Número (Number): Almacena valores númericos (enteros y decimales)

Ejemplo Arreglo (Array)
                      
        // Declaramos una variable Array
        let languages = ["HTML", "CSS", "JavaScript"];
            
        // Llamamos a la variable
        console.log(languages); // salida: ["HTML", "CSS", "JavaScript"]

        // Podemos acceder a un valor concreto del array de la siguiente manera:
        console.log(languages[2]); //salida: "JavaScript"

        //Los valores en el array comienzan desde 0

Almacena una lista ordenada de elementos.

Ejemplo Booleano (Boolean) 
                      
                    // Declaramos variables Boolean
                    let isStudent = true;
                    let isTeacher = false;
                        
                    // Llamamos a las variables
                    console.log(isStudent); // salida: true
                    console.log(isTeacher); // salida: false

                    //Otra forma de llamarlo
                    typeof(isStudent); salida: "boolean"

Representa valores verdaderos (true) o falsos (false).

Ejemplo Objeto (Object) 
                      
  // Declaramos una variable Object
  let myCv = {
  profession: "full-Stack Developer",
  year: "2025"
    };
      
  // Llamamos a la variable
  console.log(myCv); // salida: { profession: "full-Stack Developer", year: "2025" }

  //Podemos acceder a una propiedad concreta de la siguiente manera:
  console.log(myCv.profession); //salida "full-Stack-Developer"

Estructura de datos que puede contener propiedades y métodos.

Ejemplo BigInt 
                      
                // Declaramos una variable BigInt
                let PlanB = BigInt(2000000000000000);
                    
                // Llamamos a la variable
                console.log(PlanB); // salida: 2000000000000000n

Para manejar números enteros grandes.

Ejemplo Symbol
                        
        // Creamos un objeto con propiedades privadas usando Symbol
        const userProfile = Symbol('userProfile');
        const userPassword = Symbol('userPassword');
        let user = {
            name: "Carla",
            age: 29,
            [userProfile]: "carlasiles1", 
            [userPassword]: "12345" };
        
        // Llamamos a las propiedades privadas
        console.log(user[userProfile]); // salida: "carlasiles1"
        console.log(user[userPassword]); // salida: "12345"

        //El valor no se muestra si accedemos a la propiedad de forma habitual.
            console.log(user.userProfile);//salida: undefined

Identificador único utilizado para propiedades de objetos

Ejemplo Date
                      

                // Declaramos una variable Date
                let today = new Date();
                  
                // Llamamos a la variable
                console.log("Fecha actual:", today); // salida: fecha actual

Se utiliza para trabajar con fechas y horas

Ejemplo Set
                  
            // Creamos un Set
            let mySet = new Set();
                    
            // Agregar valores al Set
            mySet.add(1);
            mySet.add(2);
            mySet.add(3);
                    
            // Llamamos a la variable
            console.log(mySet); // salida: Set { 1, 2, 3 }

            //agregamos un valor repetido
            mySet.add(1);

            //llamamos a la variable
            console.log(mySet); // salida: Set { 1, 2, 3 } (no se agrega otro 1)

Almacena valores únicos, ideal para eliminar duplicados o verificar la existencia de elementos.

Ejemplo Map 
                  
              // Creamos un Map
              let myMap = new Map();
              
              // Agregamos elementos al Map
              myMap.set("userName", "Carla Siles");
              myMap.set("age", 29);
              myMap.set("profession", "Full-stack developer");
              
              // Llamamos a las claves del Map
              console.log(myMap.get("userName")); // salida: "Carla Siles"

Permite almacenar pares clave-valor donde las claves pueden ser de cualquier tipo.

Ejemplo RegExp 
                  
                    // Creamos una expresión regular para buscar palabras
                    let regex = /hello/i;
                    
                    // Usamos la expresión regular
                    let text = "Hello, world!";
                    
                    // Buscamos coincidencias
                    let result = text.match(regex);
                    
                    // Llamamos a la variable
                    console.log(result); // salida: ["Hello"]

                    //creamos la expresión regular:
                    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

                    // Ejemplo de uso:
                    const email = "carla@hotmail.com";
                    console.log(emailRegex.test(email)); // salida: true

Útil para realizar búsquedas y validaciones en texto mediante patrones.
